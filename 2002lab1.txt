/* CZ2002 Lab 1
 * This program reads a character from the user and
 * then display what type of movie fan based on the input.
 * 
 * Author: Lim Guan
 * Admin No: U1122655H
 * Group: SSP6
 * Time Written: 19th Sept 12, 10.45am
 * 
 */

import java.util.Scanner;

public class P1 {

	/**
	 * @param args
	 */
	public static void main(String[] args) {
		
		//Get input from user
		Scanner scanner = new Scanner(System.in);
		System.out.println("Enter a char: ");
		
		//Scan user input
		char choice = scanner.next().charAt(0);
		
		//Display respective info based on user's choice
		switch(choice) {
		
			//Display "Action movie fan if user inputs A/a"
			case 'A':case'a': {
				System.out.println("Action movie fan");
				break;
			}
			
			//Display "Comedy movie fan if user inputs C/c"
			case 'C':case'c': {
				System.out.println("Comedy movie fan");
				break;
			} 
			
			//Display "Drama movie fan if user inputs D/d"
			case 'D':case'd': {
				System.out.println("Drama movie fan");
				break;
			} 
			
			//Display error msg if user input is not A/a/C/c/D/d
			default: {
				System.out.println("Invalid choice");
				break;
			}
		}
			
		//Close scanner to prevent resource leakage
		scanner.close();
		
	
	}

}











/* CZ2002 Lab 1
 * This program reads a user's salary and merit points
 * then displays the grade based on the company's salary policy.
 * 
 * Author: Lim Guan
 * Admin No: U1122655H
 * Group: SSP6
 * Time Written: 19th Sept 12, 10.55am
 * 
 */

import java.util.Scanner;

public class P2 {

	/**
	 * @param args
	 */
	public static void main(String[] args) {
		
		//Get input from user
		Scanner scanner = new Scanner(System.in);
		System.out.println("Enter your salary:");
		
		//Scan user salary 
		int salary = scanner.nextInt();
		
		//Scan user merit points
		System.out.println("Enter your merit points: ");
		int merit = scanner.nextInt();
		
		//Grade C if salary is between 500 - 599, no need to look at merit points
		if (salary >= 500 && salary <= 599){
			System.out.println("Grade: C");
		}
		//Determine grade based on merit points for salary between 600 - 649
		else if (salary >= 600 && salary <= 649) {
			//Grade C is merit points is below 10
			if (merit < 10){
				System.out.println("Grade: C");
			}
			else {
				System.out.println("Grade: B");
			}
		}
		//Grade B if salary is between 650 - 699, no need to look at merit points
		else if (salary >= 650 && salary <= 699) {
			System.out.println("Grade: B");
		}
		//Determine grade based on merit points for salary between 700 - 799
		else if (salary >= 700 && salary <= 799) {
			//Grade B is merit points is below 10
			if (merit < 20){
				System.out.println("Grade: B");
			}
			else {
				System.out.println("Grade: A");
			}
		}
		//Grade A if salary is between 800 - 899, no need to look at merit points
		else if (salary >= 800 && salary <= 899) {
			System.out.println("Grade: A");
		}
		
		//Close scanner to prevent resource leakage
		scanner.close();
	}

}









/* CZ2002 Lab 1
 * This program generate tables of currency conversions 
 * from Singapore dollars to US dollars using
 * for loop, while loop and do/while loop.
 * 
 * Author: Lim Guan
 * Admin No: U1122655H
 * Group: SSP6
 * Time Written: 19th Sept 12, 12.00pm
 * 
 */

import java.util.Scanner;

public class P3 {

	/**
	 * @param args
	 */
	public static void main(String[] args) {
		
		double rate = 1.82;
		
		//Get input from user
		Scanner scanner = new Scanner(System.in);
		System.out.println("Starting:");
		
		//Scan starting value
		int starting = scanner.nextInt();
		
		System.out.println("Ending:");
		//Scan ending value
		int ending = scanner.nextInt();
		
		System.out.println("Increment:");
		//Scan increment value
		int increment = scanner.nextInt();
		
		//Validates user input, starting value cannot be smaller than ending value
		if (starting>=ending) {
			System.out.println("Input error!");
		}
		else{
			
			//First display table header
			System.out.println("S$         US$");
			System.out.println("----------------");
			
			//Using for loop
			for (int i=starting; i<=ending; i++){
				//Prints out corresponding values
				System.out.printf("%d         %.2f\n", i,i*rate);
			}
			
			//First display table header
			System.out.println("S$         US$");
			System.out.println("----------------");
			
			//Declare variable to store starting value
			int a = starting;
			
			//Using while loop
			while (a<=ending){
				
				//Prints out corresponding values
				System.out.printf("%d         %.1f\n", a,a*rate);
				
				//Unlike logic in for loop, do increment based on user input
				a += increment;
			}
			
			//First display table header
			System.out.println("S$         US$");
			System.out.println("----------------");
			
			//Declare variable to store starting value
			int b = starting;
			
			//using do/while loop	
			do{
				
				//Prints out corresponding values
				System.out.printf("%d         %.1f\n", b,b*rate);
				
				//Like logic in while loop, do increment based on user input
				b += increment;
				
			}while (b <= ending);
		}
		
		//Close scanner to prevent resource leakage
		scanner.close();

	}

}









/* CZ2002 Lab 1
 * This program reads the height from a user 
 * and prints a pattern with the specified height.
 * 
 * Author: Lim Guan
 * Admin No: U1122655H
 * Group: SSP6
 * Time Written: 19th Sept 12, 11.30am
 * 
 */

import java.util.Scanner;

public class P4 {

	/**
	 * @param args
	 */
	public static void main(String[] args) {
		
		//Get input from user
		Scanner scanner = new Scanner(System.in);
		System.out.println("Enter height:");
		
		//Scan height
		int height = scanner.nextInt();
		
		//Error if user inputs 0
		if (height == 0){
			System.out.println("Error input!");
		}
		
		//Prints patterns when height >0
		else {
			
			//First for loop for each row
			for (int i=1; i<=height; i++){
				
				//Prints what's inside the row
				for (int j=1; j<=i;j++){
					
					//Switch between printing AA and BB
					if ((j-i)%2 == 0){
						System.out.print("AA");
					}
					else {
						System.out.print("BB");
					}
				}System.out.println("");
			}
		}
		
		//Close scanner to prevent resource leakage
		scanner.close();
	}

}
